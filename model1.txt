    }
    alpha ~ dnorm(0.00000E+00, 1.00000E-04)
    beta ~ dnorm(0.00000E+00, 1.00000E-04)
    logit(pi2) <- alpha + beta * 100
    ypred ~ dbin(pi2, 240)

    q.star <- (0.01 * (1 - P0)) + P0
    bmd <- (logit(q.star) - alpha)/beta
